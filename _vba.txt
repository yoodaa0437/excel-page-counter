Sub CountPagesInFolder_V2()
    Dim folderPath As String
    Dim fileName As String
    Dim filePath As String
    Dim wb As Workbook
    Dim ws As Worksheet
    Dim bookTotalPages As Long
    Dim totalAllPages As Long
    Dim pageCount As Long
    
    Dim outSheet As Worksheet
    Dim outRow As Long
    
    ' 出力先シートの準備（マクロ実行中のシート）
    Set outSheet = ThisWorkbook.ActiveSheet
    
    ' 1. C1セルからフォルダパスを取得
    folderPath = outSheet.Range("C1").Value
    
    ' パスが空、またはフォルダが存在しない場合のチェック
    If folderPath = "" Then
        MsgBox "C1セルにフォルダパスを入力してください。", vbExclamation
        Exit Sub
    End If
    
    ' フォルダパスの末尾に「\」がない場合は補完
    If Right(folderPath, 1) <> "\" Then folderPath = folderPath & "\"
    
    ' 指定したフォルダが存在するかチェック
    If Dir(folderPath, vbDirectory) = "" Then
        MsgBox "指定されたフォルダが見つかりません:" & vbCrLf & folderPath, vbCritical
        Exit Sub
    End If

    ' 結果出力エリア（3行目以降）を初期化
    ' ※C1セルのパスを消さないよう、A3セルから最終列・最終行までをクリア
    outSheet.Rows("3:" & outSheet.Rows.Count).Clear
    
    ' 2. 結果の出力は3行目からスタート（3行目にヘッダー）
    outRow = 3
    outSheet.Cells(outRow, 1).Value = "ファイル名"
    outSheet.Cells(outRow, 2).Value = "シート名"
    outSheet.Cells(outRow, 3).Value = "ページ数"
    outSheet.Range(outSheet.Cells(outRow, 1), outSheet.Cells(outRow, 3)).Interior.Color = RGB(220, 230, 241)
    outSheet.Range(outSheet.Cells(outRow, 1), outSheet.Cells(outRow, 3)).Font.Bold = True
    
    ' 実際のデータは4行目から開始
    outRow = 4

    ' 安全対策1: 画面更新停止、警告非表示、マクロ自動実行停止
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    Application.EnableEvents = False

    ' フォルダ内の .xlsx ファイルを取得
    fileName = Dir(folderPath & "*.xlsx")
    totalAllPages = 0

    Do While fileName <> ""
        
        ' 安全対策2: 一時ファイル（~$）はスキップ
        If Left(fileName, 2) <> "~$" Then
            filePath = folderPath & fileName
            bookTotalPages = 0
            
            On Error GoTo ErrorHandler

            ' 安全対策3: 読み取り専用(ReadOnly:=True)、リンク更新なし(UpdateLinks:=0)
            Set wb = Workbooks.Open(Filename:=filePath, ReadOnly:=True, UpdateLinks:=0)
            
            For Each ws In wb.Worksheets
                ' 各シートのページ数を取得
                pageCount = ws.PageSetup.Pages.Count
                
                outSheet.Cells(outRow, 1).Value = fileName
                outSheet.Cells(outRow, 2).Value = ws.Name
                outSheet.Cells(outRow, 3).Value = pageCount
                outRow = outRow + 1
                
                bookTotalPages = bookTotalPages + pageCount
            Next ws
            
            ' ファイルごとの小計
            outSheet.Cells(outRow, 1).Value = ">> " & fileName & " の合計"
            outSheet.Cells(outRow, 3).Value = bookTotalPages
            outSheet.Rows(outRow).Font.Bold = True
            outRow = outRow + 2
            
            totalAllPages = totalAllPages + bookTotalPages

            ' 安全対策4: 保存せずに閉じる
            wb.Close SaveChanges:=False
            
            On Error GoTo 0
        End If

NextFile:
        fileName = Dir()
    Loop

    ' 最終結果の表示
    outSheet.Cells(outRow, 1).Value = "【処理完了】フォルダ内の全ページの総数"
    outSheet.Cells(outRow, 3).Value = totalAllPages
    outSheet.Rows(outRow).Font.Bold = True
    outSheet.Rows(outRow).Font.Color = vbRed

ExitRoutine:
    ' Excel設定を元に戻す
    Application.ScreenUpdating = True
    Application.DisplayAlerts = True
    Application.EnableEvents = True
    
    If totalAllPages > 0 Or outRow > 4 Then
        MsgBox "処理が完了しました。" & vbCrLf & "総ページ数: " & totalAllPages & " ページ", vbInformation
    Else
        MsgBox "対象となる .xlsx ファイルが見つかりませんでした。", vbExclamation
    End If
    Exit Sub

ErrorHandler:
    outSheet.Cells(outRow, 1).Value = "[エラー] " & fileName
    outSheet.Cells(outRow, 2).Value = Err.Description
    outSheet.Rows(outRow).Font.Color = vbRed
    outRow = outRow + 2
    
    If Not wb Is Nothing Then
        wb.Close SaveChanges:=False
    End If
    Resume NextFile
    
End Sub
